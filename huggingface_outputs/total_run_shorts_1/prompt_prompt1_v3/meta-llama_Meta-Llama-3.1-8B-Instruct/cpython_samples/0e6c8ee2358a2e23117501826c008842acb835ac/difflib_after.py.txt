import re

def IS_LINE_JUNK(line, pat=re.compile(r"\s*(?:#\s*)?$").match):
    r"""
    Return 1 for ignorable line: iff `line` is blank or contains a single '#'.

    Examples:

    >>> IS_LINE_JUNK("")
    1
    >>> IS_LINE_JUNK("#")
    1
    >>> IS_LINE_JUNK("##")
    0
    >>> IS_LINE_JUNK("a")
    0
    """
    if pat(line):
        return 1
    else:
        return 0
