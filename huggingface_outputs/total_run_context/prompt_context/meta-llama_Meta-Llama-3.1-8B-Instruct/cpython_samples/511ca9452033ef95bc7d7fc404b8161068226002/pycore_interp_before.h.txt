import requests
import json

def get_data(url):
    try:
        response = requests.get(url)
        response.raise_for_status()
    except requests.exceptions.RequestException as e:
        print(f"Error: {e}")
        return None

    try:
        data = json.loads(response.text)
        return data
    except json.JSONDecodeError as e:
        print(f"Error: {e}")
        return None

url = "https://example.com/api/data"
data = get_data(url)
